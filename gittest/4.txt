public class NaturalNumbers {

    public static void main(String[] args) {
        int n = 10; // Number of natural numbers to print and sum

        // Print natural numbers from 1 to n
        System.out.println("First " + n + " natural numbers:");
        printNaturalNumbers(n);

        // Calculate and print the sum of first n natural numbers
        int sum = calculateSum(n);
        System.out.println("Sum of first " + n + " natural numbers: " + sum);
    }

    // Method to print first n natural numbers
    public static void printNaturalNumbers(int n) {
        for (int i = 1; i <= n; i++) {
            System.out.print(i + " ");
        }
        System.out.println(); // Move to the next line after printing numbers
    }

    // Method to calculate sum of first n natural numbers
    public static int calculateSum(int n) {
        int sum = 0;
        for (int i = 1; i <= n; i++) {
            sum += i;
        }
        return sum;
    }
}
